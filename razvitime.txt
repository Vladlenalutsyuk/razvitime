DROP DATABASE IF EXISTS razvitime;
CREATE DATABASE razvitime CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE razvitime;

SET NAMES utf8mb4;
SET FOREIGN_KEY_CHECKS = 0;

-- =========================
-- 1. Пользователи
-- =========================
CREATE TABLE users (
 id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
 phone VARCHAR(30) NULL,
 email VARCHAR(255) NULL,
 password_hash VARCHAR(255) NOT NULL,
 role ENUM('parent', 'center_admin', 'admin') NOT NULL,
 status ENUM('active', 'blocked', 'pending') ЗНАЧЕНИЕ NOT NULL ПО УМОЛЧАНИЮ "активно", 
 first_login_required TINYINT(1) NOT NULL ПО УМОЛЧАНИЮ 0, 
 созданная ВРЕМЕННАЯ МЕТКА NOT NULL ПО УМОЛЧАНИЮ CURRENT_TIMESTAMP, 
 обновленная ВРЕМЕННАЯ МЕТКА NOT NULL ПО УМОЛЧАНИЮ CURRENT_TIMESTAMP ПРИ ОБНОВЛЕНИИ CURRENT_TIMESTAMP, 
 last_login_at ВРЕМЕННАЯ МЕТКА NULL ПО УМОЛЧАНИЮ NULL, 
 УНИКАЛЬНЫЙ КЛЮЧ uq_users_phone (телефон),
 УНИКАЛЬНЫЙ КЛЮЧ uq_users_email (электронная почта)
) ДВИЖОК=InnoDB;

-- =========================
-- 2. Профили родителей
-- =========================
CREATE TABLE parent_profiles (
 id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
 user_id BIGINT UNSIGNED NOT NULL,
 full_name VARCHAR(255) NOT NULL,
 city VARCHAR(150) NULL,
 telegram VARCHAR(100) NULL,
 whatsapp VARCHAR(100) NULL,
 email VARCHAR(255) NULL,
 avatar_url VARCHAR(500) NULL,
 preferred_contact ENUM('phone', 'telegram', 'whatsapp', 'email') NULL,
 notifications_enabled TINYINT(1) ЗНАЧЕНИЕ NOT NULL ПО УМОЛЧАНИЮ 1, 
 созданная ВРЕМЕННАЯ МЕТКА NOT NULL По УМОЛЧАНИЮ CURRENT_TIMESTAMP, 
 обновленная ВРЕМЕННАЯ МЕТКА NOT NULL ПО УМОЛЧАНИЮ CURRENT_TIMESTAMP ПРИ ОБНОВЛЕНИИ CURRENT_TIMESTAMP,
 ОГРАНИЧЕНИЕ fk_parent_profiles_user
 ВНЕШНИЙ КЛЮЧ (user_id) ССЫЛАЕТСЯ НА пользователей (id)
 При КАСКАДНОМ УДАЛЕНИИ,
 УНИКАЛЬНЫЙ КЛЮЧ uq_parent_profiles_user_id (user_id)
) ДВИЖОК=InnoDB;

-- =========================
-- 3. Дети
-- =========================
CREATE TABLE children (
 id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
 parent_id BIGINT UNSIGNED NOT NULL,
 name VARCHAR(255) NOT NULL,
 birthdate DATE NOT NULL,
 gender ENUM('boy', 'girl', 'other') NULL,
 photo_url VARCHAR(500) NULL,
 created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
 updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
 CONSTRAINT fk_children_parent
 FOREIGN KEY (parent_id) REFERENCES parent_profiles(id)
 ON DELETE CASCADE,
 ИДЕНТИФИКАТОР ИНДЕКСА x_children_parent_id (parent_id)
) ДВИЖОК=InnoDB;

-- =========================
-- 4. Центры
-- =========================
СОЗДАТЬ центры ТАБЛИЦ ( 
 идентификатор BIGINT БЕЗ ЗНАКА AUTO_INCREMENT PRIMARY KEY, 
 user_id BIGINT БЕЗ ЗНАКА NOT NULL, 
 имя VARCHAR(255) НЕ NULL, 
 значение параметра short_description (500) РАВНО НУЛЮ, 
 значение параметра full_description TEXT РАВНО НУЛЮ, 
 значение параметра city (150) РАВНО НУЛЮ, 
 значение параметра address(255) РАВНО НУЛЮ, 
 параметр ориентира VARCHAR (255) РАВЕН НУЛЮ, 
 параметр телефона VARCHAR (30) РАВЕН НУЛЮ, 
 параметр электронной почты VARCHAR (255) РАВЕН НУЛЮ, 
 параметр веб-сайта VARCHAR (255) РАВЕН НУЛЮ, 
 параметр telegram VARCHAR (255) РАВЕН НУЛЮ, 
 значение ПЕРЕМЕННОЙ whatsapp (255) РАВНО НУЛЮ, 
 значение ПЕРЕМЕННОЙ vk (255) РАВНО НУЛЮ, 
 значение переменной logo_url (500) РАВНО НУЛЮ, 
 значение ПЕРЕМЕННОЙ photo_url (500) РАВНО НУЛЮ, 
 is_active TINYINT(1) NOT NULL DEFAULT 1,
 moderation_status ENUM('draft', 'pending', 'approved', 'rejected') NOT NULL DEFAULT 'draft',
 created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
 updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
 CONSTRAINT fk_centers_user
 FOREIGN KEY (user_id) REFERENCES users(id)
 ON DELETE CASCADE,
 УНИКАЛЬНЫЙ КЛЮЧ uq_centers_user_id (user_id),
 ИНДЕКС idx_centers_city (город),
 ИНДЕКС idx_centers_is_active (is_active)
) ДВИЖОК=InnoDB;

-- =========================
-- 5. Тарифы подписок
-- =========================
СОЗДАТЬ ТАБЛИЦУ подписок ( 
 идентификатор BIGINT БЕЗЗНАКОВЫЙ ПЕРВИЧНЫЙ КЛЮЧ AUTO_INCREMENT, 
 имя VARCHAR(100) НЕ РАВНО NULL, 
 описание VARCHAR(500) РАВНО NULL, 
 цена ДЕСЯТИЧНАЯ (10,2) НЕ РАВНА НУЛЮ, 
 period_days INT НЕ РАВЕН НУЛЮ, 
 is_active TINYINT(1) NOT NULL ПО УМОЛЧАНИЮ 1, 
 временная МЕТКА created_at NOT NULL ПО УМОЛЧАНИЮ CURRENT_TIMESTAMP 
) ДВИЖОК=InnoDB;

-- =========================
-- 6. Подписки центров
-- =========================
CREATE TABLE center_subscriptions (
 id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
 center_id BIGINT UNSIGNED NOT NULL,
 subscription_id BIGINT UNSIGNED NOT NULL,
 start_date DATE NOT NULL,
 end_date DATE NOT NULL,
 status ENUM('active', 'expired', 'cancelled') NOT NULL DEFAULT 'active',
 created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
 ОГРАНИЧЕНИЕ fk_center_subscriptions_center
 ВНЕШНИЙ КЛЮЧ (center_id) ССЫЛАЕТСЯ НА ЦЕНТРЫ (id)
 ПРИ УДАЛЕНИИ ПО КАСКАДНОМУ ПРИНЦИПУ,
 ОГРАНИЧЕНИЕ fk_center_subscriptions_subscription
 ВНЕШНИЙ КЛЮЧ (subscription_id) ССЫЛАЕТСЯ НА ПОДПИСКИ (id)
 ПРИ УДАЛЕНИИ RESTRICT, 
 ИНДЕКСИРУЙТЕ idx_center_subscriptions_center_id (center_id), 
 ИНДЕКСИРУЙТЕ idx_center_subscriptions_status (статус)
) ДВИЖОК=InnoDB;

-- =========================
-- 7. Кружки
-- =========================
CREATE TABLE activities ( 
 идентификатор BIGINT БЕЗ ЗНАКА AUTO_INCREMENT PRIMARY KEY, 
 center_id BIGINT БЕЗ ЗНАКА NOT NULL, 
 title VARCHAR(255) НЕ NULL, 
 параметр category VARCHAR(100) НЕ РАВЕН НУЛЮ, 
 значение age_min INT НЕ РАВНО НУЛЮ, 
 значение age_max INT НЕ РАВНО НУЛЮ, 
 значение short_description VARCHAR(500) РАВНО НУЛЮ, 
 текст описания РАВЕН НУЛЮ, 
 цена ДЕСЯТИЧНАЯ (10,2) НЕ равна нулю ПО УМОЛЧАНИЮ 0,00, 
 перечисление типа оплаты ('ежемесячно', 'за_лессон', 'бесплатно') NOT NULL DEFAULT 'monthly',
 capacity INT NULL,
 is_active TINYINT(1) NOT NULL DEFAULT 1,
 created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
 updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
 CONSTRAINT fk_activities_center
 FOREIGN KEY (center_id) REFERENCES centers(id)
 ON DELETE CASCADE,
 INDEX idx_activities_center_id (center_id),
 INDEX idx_activities_category (category),
 INDEX idx_activities_age (age_min, age_max),
 INDEX idx_activities_is_active (is_active)
) ДВИЖОК=InnoDB;

-- =========================
-- 8. Сессии кружков (дни/время)
-- =========================
СОЗДАЕМ ТАБЛИЦУ activity_sessions ( 
 первичный КЛЮЧ AUTO_INCREMENT id BIGINT БЕЗ ЗНАКА, 
 activity_id BIGINT БЕЗ ЗНАКА NOT NULL, 
 день недели TINYINT NOT NULL КОММЕНТАРИЙ '1=Пн ... 7=Вс', 
 время начала НЕ РАВНО нулю, 
 ВРЕМЯ окончания НЕ РАВНО НУЛЮ, 
 временная МЕТКА created_at НЕ РАВНА НУЛЮ, ТЕКУЩАЯ ВРЕМЕННАЯ МЕТКА ПО УМОЛЧАНИЮ НЕ РАВНА НУЛЮ,
 ОГРАНИЧЕНИЕ fk_activity_sessions_activity
 ВНЕШНИЙ КЛЮЧ (activity_id) ССЫЛАЕТСЯ НА действия (id)
 ПРИ КАСКАДНОМ УДАЛЕНИИ, 
 ИНДЕКСИРУЙТЕ idx_activity_sessions_activity_id (activity_id), 
 ИНДЕКСИРУЙТЕ idx_activity_sessions_weekday (день недели)
) ДВИЖОК=InnoDB;

-- =========================
-- 9. Запись на кружки
-- =========================
CREATE TABLE enrollments (
 id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
 child_id BIGINT UNSIGNED NOT NULL,
 activity_id BIGINT UNSIGNED NOT NULL,
 status ENUM('ожидает рассмотрения', 'одобрено', 'отклонено', 'отменено') NOT NULL DEFAULT 'ожидает',
 parent_comment VARCHAR(500) NULL,
 center_comment VARCHAR(500) NULL,
 created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
 updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
 ОГРАНИЧЕНИЕ fk_enrollments_child
 ВНЕШНИЙ КЛЮЧ (child_id) ССЫЛАЕТСЯ НА children(id)
 ON DELETE CASCADE,
 ОГРАНИЧЕНИЕ fk_enrollments_activity
 ВНЕШНИЙ КЛЮЧ (activity_id) ССЫЛАЕТСЯ НА activities(id)
 ПРИ КАСКАДНОМ УДАЛЕНИИ, 
 ИНДЕКСИРУЙТЕ idx_enrollments_child_id (child_id), 
 ИНДЕКСИРУЙТЕ idx_enrollments_activity_id (activity_id), 
 ИНДЕКС idx_enrollments_status (состояние)
) ДВИЖОК=InnoDB;

-- =========================
-- 10. Школьные уроки
-- =========================
CREATE TABLE school_lessons (
 id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
 child_id BIGINT UNSIGNED NOT NULL,
 weekday TINYINT NOT NULL COMMENT '1=Пн ... 7=Вс',
 номер_урока INT NULL,
 время_начала TIME NULL,
 время_окончания TIME NULL,
 предмет VARCHAR(255) NOT NULL,
 класс VARCHAR(100) NULL,
 дата_создания TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
 дата_обновления TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
 ОГРАНИЧЕНИЕ fk_school_lessons_child
 ИНОСТРАННЫЙ КЛЮЧ (child_id) ССЫЛАЕТСЯ на таблицу children(id)
 ON DELETE CASCADE,
 ИНДЕКС idx_school_lessons_child_id (child_id), 
 ИНДЕКС idx_school_lessons_weekday (день недели)
) ДВИЖОК = InnoDB;

-- =========================
-- 11. Отзывы
-- =========================
СОЗДАЙТЕ ТАБЛИЦУ reviews ( 
 идентификатор BIGINT БЕЗЗНАКОВЫЙ ПЕРВИЧНЫЙ КЛЮЧ AUTO_INCREMENT, 
 введите ENUM('parent', 'center') NOT NULL,
 author_name VARCHAR(255) NOT NULL,
 role VARCHAR(255) NULL,
 text TEXT NOT NULL,
 center_id BIGINT UNSIGNED NULL,
 is_published TINYINT(1) NOT NULL DEFAULT 1,
 created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
 CONSTRAINT fk_reviews_center
 FOREIGN KEY (center_id) REFERENCES centers(id)
 ON DELETE SET NULL,
 INDEX idx_reviews_type (type),
 ИДЕНТИФИКАТОР ИНДЕКСА x_reviews_center_id (center_id)
) ДВИЖОК =InnoDB;

-- =========================
-- 12. Уведомления
-- =========================
CREATE TABLE notifications (
 id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
 user_id BIGINT UNSIGNED NOT NULL,
 title VARCHAR(255) NOT NULL,
 message TEXT NOT NULL,
 type ENUM('system', 'enrollment', 'subscription', 'reminder') NOT NULL DEFAULT 'system',
 is_read TINYINT(1) NOT NULL DEFAULT 0,
 временная МЕТКА created_at НЕ равна НУЛЮ ТЕКУЩАЯ ВРЕМЕННАЯ МЕТКА ПО УМОЛЧАНИЮ,
 ОГРАНИЧЕНИЕ fk_notifications_user
 ВНЕШНИЙ КЛЮЧ (user_id) ССЫЛАЕТСЯ НА пользователей (id)
 ПРИ КАСКАДНОМ УДАЛЕНИИ, 
 ИНДЕКСИРУЙТЕ idx_notifications_user_id (user_id), 
 ИНДЕКСИРУЙТЕ idx_notifications_is_read (is_read)
) ДВИЖОК = InnoDB;

-- =========================
-- 13. Заявки от центров
-- =========================
CREATE TABLE center_applications (
 id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
 center_name VARCHAR(255) NOT NULL,
 contact_person VARCHAR(255) NOT NULL,
 phone VARCHAR(30) NULL,
 email VARCHAR(255) NULL,
 city VARCHAR(150) NULL,
 comment TEXT NULL,
 status ENUM('new', 'in_review', 'approved', 'rejected') ЗНАЧЕНИЕ NOT NULL ПО УМОЛЧАНИЮ 'new', 
 создано С МЕТКОЙ ВРЕМЕНИ NOT NULL По УМОЛЧАНИЮ CURRENT_TIMESTAMP, 
 обновлено С МЕТКОЙ ВРЕМЕНИ NOT NULL ПО УМОЛЧАНИЮ CURRENT_TIMESTAMP ПРИ ОБНОВЛЕНИИ CURRENT_TIMESTAMP, 
 ИНДЕКС idx_center_applications_status (состояние)
) ДВИЖОК=InnoDB;

-- =========================
-- 14. Токены сброса пароля
-- =========================
CREATE TABLE password_reset_tokens (
 id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
 user_id BIGINT UNSIGNED NOT NULL,
 token VARCHAR(255) NOT NULL,
 expires_at DATETIME NOT NULL,
 used_at DATETIME NULL,
 created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
 CONSTRAINT fk_password_reset_tokens_user
 FOREIGN KEY (user_id) REFERENCES users(id)
 ON DELETE CASCADE,
 UNIQUE KEY uq_password_reset_tokens_token (token),
 INDEX idx_password_reset_tokens_user_id (user_id)
) ENGINE=InnoDB;

УСТАНОВИТЕ FOREIGN_KEY_CHECKS = 1;
