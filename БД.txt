DROP DATABASE IF EXISTS razvitime;
CREATE DATABASE razvitime CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE razvitime;

SET NAMES utf8mb4;
SET FOREIGN_KEY_CHECKS = 0;

-- =========================
-- 1. Пользователи
-- =========================
CREATE TABLE users (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    phone VARCHAR(30) NULL,
    email VARCHAR(255) NULL,
    password_hash VARCHAR(255) NOT NULL,
    role ENUM('parent', 'center_admin', 'admin') NOT NULL,
    status ENUM('active', 'blocked', 'pending') NOT NULL DEFAULT 'active',
    first_login_required TINYINT(1) NOT NULL DEFAULT 0,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    last_login_at TIMESTAMP NULL DEFAULT NULL,
    UNIQUE KEY uq_users_phone (phone),
    UNIQUE KEY uq_users_email (email)
) ENGINE=InnoDB;

-- =========================
-- 2. Профили родителей
-- =========================
CREATE TABLE parent_profiles (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT UNSIGNED NOT NULL,
    full_name VARCHAR(255) NOT NULL,
    city VARCHAR(150) NULL,
    telegram VARCHAR(100) NULL,
    whatsapp VARCHAR(100) NULL,
    email VARCHAR(255) NULL,
    avatar_url VARCHAR(500) NULL,
    preferred_contact ENUM('phone', 'telegram', 'whatsapp', 'email') NULL,
    notifications_enabled TINYINT(1) NOT NULL DEFAULT 1,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    CONSTRAINT fk_parent_profiles_user
        FOREIGN KEY (user_id) REFERENCES users(id)
        ON DELETE CASCADE,
    UNIQUE KEY uq_parent_profiles_user_id (user_id)
) ENGINE=InnoDB;

-- =========================
-- 3. Дети
-- =========================
CREATE TABLE children (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    parent_id BIGINT UNSIGNED NOT NULL,
    name VARCHAR(255) NOT NULL,
    birthdate DATE NOT NULL,
    gender ENUM('boy', 'girl', 'other') NULL,
    photo_url VARCHAR(500) NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    CONSTRAINT fk_children_parent
        FOREIGN KEY (parent_id) REFERENCES parent_profiles(id)
        ON DELETE CASCADE,
    INDEX idx_children_parent_id (parent_id)
) ENGINE=InnoDB;

-- =========================
-- 4. Центры
-- =========================
CREATE TABLE centers (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT UNSIGNED NOT NULL,
    name VARCHAR(255) NOT NULL,
    short_description VARCHAR(500) NULL,
    full_description TEXT NULL,
    city VARCHAR(150) NOT NULL,
    address VARCHAR(255) NOT NULL,
    landmark VARCHAR(255) NULL,
    phone VARCHAR(30) NULL,
    email VARCHAR(255) NULL,
    website VARCHAR(255) NULL,
    telegram VARCHAR(255) NULL,
    whatsapp VARCHAR(255) NULL,
    vk VARCHAR(255) NULL,
    logo_url VARCHAR(500) NULL,
    photo_url VARCHAR(500) NULL,
    is_active TINYINT(1) NOT NULL DEFAULT 1,
    moderation_status ENUM('draft', 'pending', 'approved', 'rejected') NOT NULL DEFAULT 'draft',
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    CONSTRAINT fk_centers_user
        FOREIGN KEY (user_id) REFERENCES users(id)
        ON DELETE CASCADE,
    UNIQUE KEY uq_centers_user_id (user_id),
    INDEX idx_centers_city (city),
    INDEX idx_centers_is_active (is_active)
) ENGINE=InnoDB;

-- =========================
-- 5. Тарифы подписок
-- =========================
CREATE TABLE subscriptions (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    description VARCHAR(500) NULL,
    price DECIMAL(10,2) NOT NULL,
    period_days INT NOT NULL,
    is_active TINYINT(1) NOT NULL DEFAULT 1,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB;

-- =========================
-- 6. Подписки центров
-- =========================
CREATE TABLE center_subscriptions (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    center_id BIGINT UNSIGNED NOT NULL,
    subscription_id BIGINT UNSIGNED NOT NULL,
    start_date DATE NOT NULL,
    end_date DATE NOT NULL,
    status ENUM('active', 'expired', 'cancelled') NOT NULL DEFAULT 'active',
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_center_subscriptions_center
        FOREIGN KEY (center_id) REFERENCES centers(id)
        ON DELETE CASCADE,
    CONSTRAINT fk_center_subscriptions_subscription
        FOREIGN KEY (subscription_id) REFERENCES subscriptions(id)
        ON DELETE RESTRICT,
    INDEX idx_center_subscriptions_center_id (center_id),
    INDEX idx_center_subscriptions_status (status)
) ENGINE=InnoDB;

-- =========================
-- 7. Кружки
-- =========================
CREATE TABLE activities (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    center_id BIGINT UNSIGNED NOT NULL,
    title VARCHAR(255) NOT NULL,
    category VARCHAR(100) NOT NULL,
    age_min INT NOT NULL,
    age_max INT NOT NULL,
    short_description VARCHAR(500) NULL,
    description TEXT NULL,
    price DECIMAL(10,2) NOT NULL DEFAULT 0.00,
    payment_type ENUM('monthly', 'per_lesson', 'free') NOT NULL DEFAULT 'monthly',
    capacity INT NULL,
    is_active TINYINT(1) NOT NULL DEFAULT 1,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    CONSTRAINT fk_activities_center
        FOREIGN KEY (center_id) REFERENCES centers(id)
        ON DELETE CASCADE,
    INDEX idx_activities_center_id (center_id),
    INDEX idx_activities_category (category),
    INDEX idx_activities_age (age_min, age_max),
    INDEX idx_activities_is_active (is_active)
) ENGINE=InnoDB;

-- =========================
-- 8. Сессии кружков (дни/время)
-- =========================
CREATE TABLE activity_sessions (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    activity_id BIGINT UNSIGNED NOT NULL,
    weekday TINYINT NOT NULL COMMENT '1=Пн ... 7=Вс',
    start_time TIME NOT NULL,
    end_time TIME NOT NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_activity_sessions_activity
        FOREIGN KEY (activity_id) REFERENCES activities(id)
        ON DELETE CASCADE,
    INDEX idx_activity_sessions_activity_id (activity_id),
    INDEX idx_activity_sessions_weekday (weekday)
) ENGINE=InnoDB;

-- =========================
-- 9. Записи на кружки
-- =========================
CREATE TABLE enrollments (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    child_id BIGINT UNSIGNED NOT NULL,
    activity_id BIGINT UNSIGNED NOT NULL,
    status ENUM('pending', 'approved', 'declined', 'cancelled') NOT NULL DEFAULT 'pending',
    parent_comment VARCHAR(500) NULL,
    center_comment VARCHAR(500) NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    CONSTRAINT fk_enrollments_child
        FOREIGN KEY (child_id) REFERENCES children(id)
        ON DELETE CASCADE,
    CONSTRAINT fk_enrollments_activity
        FOREIGN KEY (activity_id) REFERENCES activities(id)
        ON DELETE CASCADE,
    INDEX idx_enrollments_child_id (child_id),
    INDEX idx_enrollments_activity_id (activity_id),
    INDEX idx_enrollments_status (status)
) ENGINE=InnoDB;

-- =========================
-- 10. Школьные уроки
-- =========================
CREATE TABLE school_lessons (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    child_id BIGINT UNSIGNED NOT NULL,
    weekday TINYINT NOT NULL COMMENT '1=Пн ... 7=Вс',
    lesson_number INT NULL,
    start_time TIME NULL,
    end_time TIME NULL,
    subject VARCHAR(255) NOT NULL,
    classroom VARCHAR(100) NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    CONSTRAINT fk_school_lessons_child
        FOREIGN KEY (child_id) REFERENCES children(id)
        ON DELETE CASCADE,
    INDEX idx_school_lessons_child_id (child_id),
    INDEX idx_school_lessons_weekday (weekday)
) ENGINE=InnoDB;

-- =========================
-- 11. Отзывы
-- =========================
CREATE TABLE reviews (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    type ENUM('parent', 'center') NOT NULL,
    author_name VARCHAR(255) NOT NULL,
    role VARCHAR(255) NULL,
    text TEXT NOT NULL,
    center_id BIGINT UNSIGNED NULL,
    is_published TINYINT(1) NOT NULL DEFAULT 1,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_reviews_center
        FOREIGN KEY (center_id) REFERENCES centers(id)
        ON DELETE SET NULL,
    INDEX idx_reviews_type (type),
    INDEX idx_reviews_center_id (center_id)
) ENGINE=InnoDB;

-- =========================
-- 12. Уведомления
-- =========================
CREATE TABLE notifications (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT UNSIGNED NOT NULL,
    title VARCHAR(255) NOT NULL,
    message TEXT NOT NULL,
    type ENUM('system', 'enrollment', 'subscription', 'reminder') NOT NULL DEFAULT 'system',
    is_read TINYINT(1) NOT NULL DEFAULT 0,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_notifications_user
        FOREIGN KEY (user_id) REFERENCES users(id)
        ON DELETE CASCADE,
    INDEX idx_notifications_user_id (user_id),
    INDEX idx_notifications_is_read (is_read)
) ENGINE=InnoDB;

-- =========================
-- 13. Заявки от центров
-- =========================
CREATE TABLE center_applications (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    center_name VARCHAR(255) NOT NULL,
    contact_person VARCHAR(255) NOT NULL,
    phone VARCHAR(30) NULL,
    email VARCHAR(255) NULL,
    city VARCHAR(150) NULL,
    comment TEXT NULL,
    status ENUM('new', 'in_review', 'approved', 'rejected') NOT NULL DEFAULT 'new',
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_center_applications_status (status)
) ENGINE=InnoDB;

-- =========================
-- 14. Токены сброса пароля
-- =========================
CREATE TABLE password_reset_tokens (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT UNSIGNED NOT NULL,
    token VARCHAR(255) NOT NULL,
    expires_at DATETIME NOT NULL,
    used_at DATETIME NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_password_reset_tokens_user
        FOREIGN KEY (user_id) REFERENCES users(id)
        ON DELETE CASCADE,
    UNIQUE KEY uq_password_reset_tokens_token (token),
    INDEX idx_password_reset_tokens_user_id (user_id)
) ENGINE=InnoDB;

SET FOREIGN_KEY_CHECKS = 1;
